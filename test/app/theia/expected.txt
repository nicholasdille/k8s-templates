apiVersion: extensions/v1beta1
kind: Deployment
metadata:
  name: theia
  namespace: theia
spec:
  replicas: 1
  selector:
    matchLabels:
      app: theia
  template:
    metadata:
      labels:
        app: theia
    spec:
      containers:
      - name: theia
        image: theiaide/theia:latest
        imagePullPolicy: Always
        ports:
        - name: web
          containerPort: 3000

metadata:
  name: theia
  namespace: theia
spec:
  entryPoints:
  - http
  routes:
  - match: Host(`ide.example.com`)
    kind: Rule
    priority: 12
    services:
    - name: theia
      port: 80
      strategy: RoundRobin
---

metadata:
  name: theia
  metadata:
    labels:
      name: theia
---
kind: Service
apiVersion: v1
metadata:
  name: theia
  namespace: theia
spec:
  selector:
    app: theia
  ports:
  - name: web
    port: 80
    targetPort: 3000
