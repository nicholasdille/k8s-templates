apiVersion: cert-manager.io/v1alpha2
kind: Certificate
metadata:
  name: nginx
spec:
  secretName: certificate-nginx
  duration: 2160h
  renewBefore: 360h
  commonName: nginx.example.io
  isCA: false
  keySize: 2048
  keyAlgorithm: rsa
  keyEncoding: pkcs1
  usages:
  - server auth
  - client auth
  dnsNames:
  - nginx.example.io
  issuerRef:
    name: letsencrypt-staging
    kind: ClusterIssuer
---
apiVersion: v1
kind: ConfigMap
metadata:
  name: nginx
data:
  index.html: |
    <html>
    <head><title>Test</title></head>
    <body><h1>Test</h1>Test</body>
    </html>
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx
spec:
  replicas: 1
  selector:
    matchLabels:
      pod: nginx
  template:
    metadata:
      labels:
        pod: nginx
    spec:
      volumes:
      - name: data
        configMap:
          name: nginx
      - name: tz-config
        hostPath:
          path: /usr/share/zoneinfo/Europe/Berlin
      containers:
      - name: main
        image: nginx:latest
        ports:
        - containerPort: 80
        volumeMounts:
        - name: data
          mountPath: /var/share/nginx/html
        - name: tz-config
          mountPath: /etc/localtime
---
apiVersion: externaldns.k8s.io/v1alpha1
kind: DNSEndpoint
metadata:
  name: nginx
spec:
  endpoints:
  - dnsName: nginx.example.io
    recordTTL: 180
    recordType: CNAME
    targets:
    - lb.example.io
---
apiVersion: traefik.containo.us/v1alpha1
kind: IngressRoute
metadata:
  name: nginx
  labels:
    example.io/loadbalancer: default
spec:
  entryPoints:
  - http
  routes:
  - match: Host(`nginx.example.io`)
    kind: Rule
    priority: 12
    services:
    - name: nginx
      port: 80
      strategy: RoundRobin
---
apiVersion: v1
kind: Service
metadata:
  name: nginx
spec:
  type: ClusterIP
  selector:
    pod: nginx
  ports:
  - protocol: TCP
    port: 80
    targetPort: 80
