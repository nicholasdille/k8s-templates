apiVersion: apps/v1
kind: Deployment
metadata:
  name: registry
  namespace: registry
spec:
  replicas: 1
  selector:
    matchLabels:
      pod: registry
  template:
    metadata:
      labels:
        pod: registry
    spec:
      containers:
      - name: main
        image: registry:2
        imagePullPolicy: Always
        ports:
        - name: api
          containerPort: 5000
        volumeMounts:
        - name: registry-data
          mountPath: /var/lib/registry
      volumes:
      - name: registry-data
        hostPath:
          path: /data/registry-data
          type: DirectoryOrCreate
---
kind: Service
apiVersion: v1
metadata:
  name: registry
  namespace: registry
spec:
  selector:
    pod: registry
  ports:
  - name: api
    port: 5000
    targetPort: 5000
---
apiVersion: traefik.containo.us/v1alpha1
kind: IngressRoute
metadata:
  name: registry
  namespace: registry
spec:
  entryPoints:
  - http
  routes:
  - match: Host(`reg.example.com`) && PathPrefix(`/v2`)
    kind: Rule
    priority: 12
    services:
    - name: registry
      port: 5000
      strategy: RoundRobin
---
apiVersion: apps/v1
kind: Deployment
metadata:
  name: registry-web
  namespace: registry
spec:
  replicas: 1
  selector:
    matchLabels:
      app: registry-web
  template:
    metadata:
      labels:
        app: registry-web
    spec:
      containers:
      - name: registry-web
        image: konradkleine/docker-registry-frontend:v2
        imagePullPolicy: Always
        ports:
        - name: web
          containerPort: 80
        env:
        - name: ENV_DOCKER_REGISTRY_HOST
          value: registry
        - name: ENV_DOCKER_REGISTRY_PORT
          value: "5000"
---
